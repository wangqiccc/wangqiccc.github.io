利用http模块搭建简单服务器

Posted on 2018-03-27

可以处理文件请求和ajax请求

所需模块及方法

内部模块

  • request
    • request(url,callback) 请求某个地址 在回调函数总处理数据
  • http
    • createServer 创建一个服务器
  • path
    • resolve 把一个路径或路径片段的序列解析为一个绝对路径
  • fs
    • readFileSync 同步读取文件
  • queryString
    • parse 将查询字符串序列化为对象
  • url
    • parse 将请求url序列化

    $("#search").click(function () {
        let val=$("#date").val();

    function getTable(val) {
        $.get("/api/setDate", {val}, function (r) {
            if (!r) {
                alert('暂无数据');
            }
            let data=r.data[val];
            console.log(data);
            let str="";
            data.forEach(function (obj, index) {
                str+=`
                <tr>
                <td>${obj.startTime}</td>
                <td>${obj.leftName}</td>
                <td>${obj.leftGoal}</td>
                <td>${obj.rightGoal}</td>
                <td>${obj.rightName}</td>
                <td><a href="${obj.latestNews.url}" target="_blank">战报</a></td>
                </tr>
              `;
            });
            $("tbody").html(str);
        })
    }
    function setZero(val) {
        return val < 10 ? "0" + val : val;
    }
    let date=new Date();
    let message=date.getFullYear() + "-" + setZero(date.getMonth() + 1) + "-" + setZero(date.getDate());
    getTable(message);
